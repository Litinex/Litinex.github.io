Python中if elif else用法

#条件判断：如果分数超过680，我就去清华读书
#在python中if所管辖的区域用代码块的缩进来划分
score = 700
if score >680:
    print("欢迎你来清华读书")
    print("也恭喜你即将踏入精彩的大学生活")
print("-------------------------")

#B站账号密码输入判断的逻辑案例
ok_account = "188888"
ok_password = "666666"
account = input("请输入B站账号：")
password = input("请输入B站密码：")

if account == ok_account and password == ok_password:
    print("登录成功")
else:
    print("登录失败")

#根据用户输入的数字，判断该数字是正数还是负数
num = int(input("请输入数字 "))

if num > 0:
    print(f"{num} 是一个正数")

elif num < 0:
    print(f"{num} 是一个负数")
else:
    print(f"{num} 是0")
