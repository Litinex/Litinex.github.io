Python中的运算符

print("10 + 4 = ", 10 + 4) #加 
print("10 - 4 = ", 10 - 4) #减
print("10 * 4 = ", 10 * 4) #乘
print("10 / 4 = ", 10 / 4) #除
print("10 // 4 = ", 10 // 4) #整除
print("10 % 4 = ", 10 % 4) #取余/求模
print("10 ** 4 = ", 10 ** 4) #幂指数

#案例1：要求输入两个数x与y，分别输出x+y的结果和x-y的结果
#注意：由于计算机底层是基于二进制来进行数据的存储和处理的，而二进制是无法精确表示所有的小数的，因此涉及到浮点数的运算时，可能损失精度。
x = input()
y = input()
print(float(x) + float(y), float(x) - float(y))
