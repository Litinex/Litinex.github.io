Python中输入与输出

#获取键盘上输入的数据：input
name = input("请输入您的姓名：")
age = input("请输入您的年龄：")

print(f"您好,{name},{age}")

#需求：小智的银行卡中有10000元，现在到ATM进行取钱操作，请根据输入的金额执行取钱操作，取钱完毕后，展示其银行卡余额。
money = 10000
take_money = input("请输入您要取的金额：")
last_money = money - int(take_money)
print(f"您剩余的金额为：{last_money}")
#也可以这样写
print(f"您剩余的金额为：{money - int(take_money)}")

#需求：根据用户输入的两个数字，计算两个数之和，并将其输出到控制台。
a = input()
b = input()
print(f"{int(a)+int(b)}")
