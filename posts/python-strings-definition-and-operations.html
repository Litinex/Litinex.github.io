字符串的定义以及操作

#常见数据类型
print("Hello")
print(type("Hello"))

print(type(10))
print(type(3.14))
print(type(True))
print(type(False))
print(type(None))

#如果想知道该数据是否属于指定类型，应用isinstance()来检查
num = 5.0
print(num)
print(isinstance(num,int))
#返回的是布尔值

#字符串
#定义字符串的三种方式
s1 = "Hello"#双引号定义方式
s2 = 'Python'#单引号定义方式
s3 = """
Hello:
    1234567
    2345678
"""#三引号（多行字符串）
print(s1)
print(s2)
print(s3)

print(type(s1),type(s2),type(s3))

#定义字符串(含转义字符)  \n换行  \t制表符
msg = 'It\'s very good'
print(msg)
msg1 = "Hello 的意思就是\"您好\""
msg1 = 'Hello 的意思就是"您好"'

print("\t123\n\t345")#换行符，制表符相当于tab缩进

#字符串的拼接 
s1 = "人生苦短" "我用Python"       ",OK"
print(s1)

msg1 = "人生苦短"
msg2 = "我用Python"
print("龟叔说：" + msg1 + "," + msg2)

#案例：根据自己的实际情况，输出个人的详细信息
# str(int数字)将int类型的数字转换成字符串
name = "涛哥"
age = 18
pro = "软件工程"
hobby = "Python Java"
print("大家好,我是" + name + ",今年" + str(age) + "岁，学习的专业是" + pro + ",爱好" + hobby)

#字符串格式化：占位符%s
name = "涛哥"
age = 18
pro = "软件工程"
hobby = "Python Java"
print("大家好,我是 %s ,今年 %s 岁，学习的专业是 %s ,爱好 %s" %(name,age,pro,hobby))

#字符串格式化：f"内容{变量/表达式}"，f不能省
name = "叶子庄"
print(f"大家好，我是{name}，今年{age}岁")
